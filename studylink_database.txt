CREATE TABLE USUARIO(
	ID INT AUTO_INCREMENT PRIMARY KEY,
	NOMBRE VARCHAR(30) NOT NULL,
	APELLIDO VARCHAR(30) NOT NULL,
	EMAIL VARCHAR(40) NOT NULL,
	NUMERO_TELEFONO VARCHAR (13)
);

CREATE TABLE GESTOR(
	ID INT AUTO_INCREMENT PRIMARY KEY,
	NOMBRE VARCHAR(30) NOT NULL,
	APELLIDO VARCHAR(30) NOT NULL,
	EMAIL VARCHAR(40) NOT NULL,
	NUMERO_TELEFONO VARCHAR (13)
);

CREATE TABLE TRAMITE(
	ID INT AUTO_INCREMENT PRIMARY KEY,
	ID_USUARIO INT,
	ID_GESTOR INT,
	CODIGO_ACCESO VARCHAR(12) UNIQUE,
	TIPO_TRAMITE VARCHAR(20),
	ESTADO VARCHAR(15),
	DESCRIPCION VARCHAR(200),
	FECHA_INICIO DATETIME,
	FECHA_ACTUALIZACION DATETIME,
	CONSTRAINT FK_TRAMITE_ID_USUARIO FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO(ID),
	CONSTRAINT FK_TRAMITE_ID_GESTOR FOREIGN KEY (ID_GESTOR) REFERENCES GESTOR(ID)
);
